        document.addEventListener("DOMContentLoaded", () => {

               0. LENIS SMOOTH SCROLL INIT

                direction: 'vertical',
                gestureDirection: 'vertical',
                smooth: true,
                mouseMultiplier: 1,
                smoothTouch: false,
                touchMultiplier: 2,
                infinite: false,
            });

            lenis.on('scroll', ScrollTrigger.update);

            gsap.ticker.add((time)=>{
                lenis.raf(time * 1000);
            });

            gsap.ticker.lagSmoothing(0, 0);

            // Register ScrollTrigger
            gsap.registerPlugin(ScrollTrigger);

            /* ==========================================
               1. FULLSCREEN MENU ANIMATION
            ========================================== */
            const menuToggle = document.getElementById('menu-toggle');
            const menuText = document.getElementById('menu-text');
            const menuIconBurger = document.getElementById('menu-icon-burger');
            const menuIconClose = document.getElementById('menu-icon-close');
            const mobileMenu = document.getElementById('mobile-menu');
            const menuItems = document.querySelectorAll('.menu-item');
            const menuInfo = document.querySelector('.menu-info');
            const desktopNav = document.getElementById('desktop-nav');
            
            let isMenuOpen = false;

            menuToggle.addEventListener('click', () => {
                isMenuOpen = !isMenuOpen;
                
                if (isMenuOpen) {
                    // Open Menu Overlay
                    mobileMenu.classList.remove('opacity-0', 'pointer-events-none');
                    mobileMenu.classList.add('opacity-100', 'pointer-events-auto');
                    
                    // Stop scrolling when menu is open
                    lenis.stop();

                    // Hide Desktop Nav
                    if(desktopNav) gsap.to(desktopNav, { opacity: 0, y: -20, duration: 0.4, pointerEvents: 'none' });

                    // Button UI Update
                    menuText.innerText = "CLOSE";
                    menuIconBurger.classList.add('hidden');
                    menuIconClose.classList.remove('hidden');
                    
                    // Stagger reveal text links
                    gsap.to(menuItems, { y: "0%", duration: 0.8, stagger: 0.05, ease: "power4.out", delay: 0.1 });
                    if(menuInfo) gsap.to(menuInfo, { y: 0, opacity: 1, duration: 0.8, ease: "power4.out", delay: 0.4 });

                } else {
                    // Close Menu Overlay
                    mobileMenu.classList.add('opacity-0', 'pointer-events-none');
                    mobileMenu.classList.remove('opacity-100', 'pointer-events-auto');
                    
                    // Restore scrolling
                    lenis.start();

                    // Show Desktop Nav
                    if(desktopNav) gsap.to(desktopNav, { opacity: 1, y: 0, duration: 0.4, pointerEvents: 'auto', delay: 0.2 });

                    // Button UI Update
                    menuText.innerText = "MENU";
                    menuIconBurger.classList.remove('hidden');
                    menuIconClose.classList.add('hidden');
                    
                    // Hide links for next time
                    gsap.to(menuItems, { y: "120%", duration: 0.4, ease: "power2.in" });

                }
            });

            /* ==========================================
               2. INITIAL ENTRANCE FADE-IN
            ========================================== */
            gsap.fromTo(".gs-reveal", 
                { y: 30, opacity: 0 }, 
                { y: 0, opacity: 1, duration: 1, ease: "power3.out", delay: 0.1 }
            );

            /* ==========================================
               3. MAGNETIC BUTTON EFFECT
            ========================================== */
            const magneticWraps = document.querySelectorAll('.magnetic-wrap');
            
            magneticWraps.forEach((wrap) => {
                const btn = wrap.querySelector('.magnetic-btn');
                
                wrap.addEventListener('mousemove', (e) => {
                    const rect = wrap.getBoundingClientRect();
                    const h = rect.width / 2;
                    const w = rect.height / 2;
                    // Calculate distance from center of the wrapper
                    const x = e.clientX - rect.left - h;
                    const y = e.clientY - rect.top - w;

                    // Move the button towards the mouse slightly
                    gsap.to(btn, { 
                        x: x * 0.4, 
                        y: y * 0.4, 
                        duration: 0.4,

                    });

                    // Snap back to original position
                    gsap.to(btn, { 
                        x: 0, 
                        y: 0, 
                        duration: 0.7, 
                        ease: 'elastic.out(1, 0.3)' 
                    });
                });

                gsap.fromTo(stat, 
                    { innerHTML: 0, opacity: 0, y: 30 }, 
                    {
                        innerHTML: target, 
                        opacity: 1, 
                        y: 0, 
                        duration: 2.5, 
                        ease: "expo.out",
                        delay: index * 0.1,
                        scrollTrigger: { 
                            trigger: ".gs-about-stats-container", 
                            start: "top 85%" 
                        },
                        modifiers: {
                            innerHTML: function(i) { 
                                // Format logic for decimals vs integers
                                let val = parseFloat(i);
                                let formatted = target % 1 !== 0 ? val.toFixed(1) : Math.floor(val);
                                return prefix + formatted + suffix; 
                            }
                        }

                    return horizContainer.scrollWidth - window.innerWidth;

                const horizTl = gsap.timeline({
                    scrollTrigger: {
                        trigger: horizWrapper,
                        start: "top top",
                        end: () => `+=${getScrollAmount()}`,
                        pin: true,
                        scrub: 1,
                        invalidateOnRefresh: true,
                        anticipatePin: 1
                    }
                });

                // Track movement
                horizTl.to(horizContainer, {
                    x: () => -getScrollAmount(),
                    ease: "none"
                }, 0);

                // Perfectly synced reverse parallax for images
                horizImages.forEach((img) => {
                    horizTl.to(img, {
                        xPercent: 12, // Smooth percentage-based shift within the bounded image
                        ease: "none"
                    }, 0); 
                });

                horizWrapper.addEventListener('touchstart', (e) => {
                    startX = e.touches[0].clientX;
                    startY = e.touches[0].clientY;
                }, {passive: true});

                horizWrapper.addEventListener('touchmove', (e) => {
                    if (!startX || !startY) return;
                    let diffX = startX - e.touches[0].clientX;
                    let diffY = startY - e.touches[0].clientY;
                    
                    // If horizontal swipe is more prominent than vertical, drive the scroll
                    if (Math.abs(diffX) > Math.abs(diffY) && Math.abs(diffX) > 5) {
                        window.scrollBy({ top: diffX * 1.5, left: 0, behavior: 'auto' });
                        startX = e.touches[0].clientX; 
                        startY = e.touches[0].clientY;
                    }
                }, {passive: true});

                        // Calculate width of one card + gap to accurately scroll one item
                        const gap = parseFloat(window.getComputedStyle(horizContainer).gap) || 0;
                        return projectCards[0].offsetWidth + gap;

                        lenis.scrollTo(window.scrollY + getCardScrollAmount(), { duration: 1.2, lock: false });
                    });

                    btnPrev.addEventListener('click', () => {
                        lenis.scrollTo(window.scrollY - getCardScrollAmount(), { duration: 1.2, lock: false });
                    });
                }

            function animateTestimonialChange(newIndex, direction) {
                if(isAnimating) return;

                const elements = [testiHeadline, testiText, testiName, testiRole, testiCaption];

                    opacity: 0,
                    y: direction === 'next' ? -20 : 20,
                    duration: 0.4,
                    stagger: 0.05,
                    ease: "power2.in"
                }, 0);
